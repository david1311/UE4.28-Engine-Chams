## Material Dumper
```cpp
// Use this code to dump/find cham materials to use for whatever UE game your doing!
// This material dumper was generated by gemini 3.5 flash.
void DumpMaterials()
{
    // Define an output path (e.g., your Documents folder or game directory)
    std::ofstream dumpFile("MaterialDump.txt", std::ios::out | std::ios::trunc);
    if (!dumpFile.is_open()) {
        // Failed to open file
        return;
    }

    dumpFile << "=== Unreal Engine 4 Material Dump ===\n";
    dumpFile << "Total Objects to check: " << GObjects->NumElements << "\n\n";

    int32_t materialCount = 0;

    for (int32_t i = 0; i < GObjects->NumElements; ++i)
    {
        FUObjectItem* objectItem = &GObjects->Objects[i];
        if (!objectItem || !objectItem->Object) 
            continue;

        UObject* obj = objectItem->Object;

        // Ensure the object isn't pending kill
        // (Adjust flags mask depending on the specific UE4 version, usually 0x4000)
        if (objectItem->Flags & 0x4000) 
            continue; 

        // Get the full name (e.g., "Material /Engine/EngineMaterials/DefaultMaterial.DefaultMaterial")
        std::string fullName = obj->GetFullName();

        // Filter for Materials and Material Instances
        if (fullName.find("Material ") != std::string::npos || 
            fullName.find("MaterialInstanceConstant ") != std::string::npos ||
            fullName.find("MaterialInstanceDynamic ") != std::string::npos)
        {
            dumpFile << "[" << i << "] " << fullName << "\n";
            materialCount++;
        }
    }

    dumpFile << "\n=== Dump Finished! Found " << materialCount << " materials. ===";
    dumpFile.close();
}
```
